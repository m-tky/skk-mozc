cmake_minimum_required(VERSION 3.16)
project(mozc-client-cli CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

find_package(Protobuf REQUIRED)
find_package(PkgConfig REQUIRED)
pkg_check_modules(PROTOBUF_PC REQUIRED IMPORTED_TARGET protobuf)

# Layout staged by the derivation:
#   ./mozc_client/         C++ implementation
#   ./proto/protocol/*.proto   from mozc-src
#
# protoc gets -I ./proto so that "import \"protocol/foo.proto\";" inside
# mozc's protos resolves cleanly. Generated headers land at
#   <build>/protocol/commands.pb.h
# and our C++ does  #include "protocol/commands.pb.h".

set(PROTO_DIR "${CMAKE_SOURCE_DIR}/proto")
file(MAKE_DIRECTORY "${CMAKE_BINARY_DIR}/protocol")
file(MAKE_DIRECTORY "${CMAKE_BINARY_DIR}/ipc")

set(PROTO_FILES
    protocol/commands.proto
    protocol/candidate_window.proto
    protocol/config.proto
    protocol/engine_builder.proto
    protocol/user_dictionary_storage.proto
    protocol/renderer_command.proto
    protocol/renderer_style.proto
    ipc/ipc.proto
)

set(PROTO_GENERATED)
foreach(P ${PROTO_FILES})
    get_filename_component(DIR ${P} DIRECTORY)
    get_filename_component(BASE ${P} NAME_WE)
    set(OUT_CC "${CMAKE_BINARY_DIR}/${DIR}/${BASE}.pb.cc")
    set(OUT_H  "${CMAKE_BINARY_DIR}/${DIR}/${BASE}.pb.h")
    add_custom_command(
        OUTPUT  ${OUT_CC} ${OUT_H}
        COMMAND ${Protobuf_PROTOC_EXECUTABLE}
                --cpp_out=${CMAKE_BINARY_DIR}
                -I${PROTO_DIR}
                ${PROTO_DIR}/${P}
        DEPENDS ${PROTO_DIR}/${P}
        COMMENT "protoc ${P}")
    list(APPEND PROTO_GENERATED ${OUT_CC})
endforeach()

add_executable(mozc-client-cli
    mozc_client/cli_main.cpp
    mozc_client/mozc_client.cpp
    mozc_client/ipc_socket.cpp
    log/log.cpp
    ${PROTO_GENERATED}
)
target_include_directories(mozc-client-cli PRIVATE
    "${CMAKE_SOURCE_DIR}"
    "${CMAKE_BINARY_DIR}")
target_link_libraries(mozc-client-cli PRIVATE PkgConfig::PROTOBUF_PC)

include(GNUInstallDirs)
install(TARGETS mozc-client-cli DESTINATION ${CMAKE_INSTALL_BINDIR})
