cmake_minimum_required(VERSION 3.16)
project(mozc-client-cli CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

find_package(Protobuf REQUIRED)

# Source layout:
#   ./mozc_client/      C++ implementation (from this repo)
#   ./proto/            commands.proto, config.proto (copied from mozc-src)
# The Nix derivation arranges those before invoking cmake.

set(PROTO_DIR "${CMAKE_SOURCE_DIR}/proto")
file(MAKE_DIRECTORY "${CMAKE_BINARY_DIR}/proto")

protobuf_generate_cpp(PROTO_SRCS PROTO_HDRS
    "${PROTO_DIR}/commands.proto"
    "${PROTO_DIR}/config.proto")

add_executable(mozc-client-cli
    mozc_client/cli_main.cpp
    mozc_client/mozc_client.cpp
    mozc_client/ipc_socket.cpp
    ${PROTO_SRCS}
)
target_include_directories(mozc-client-cli PRIVATE
    "${CMAKE_SOURCE_DIR}"
    "${CMAKE_BINARY_DIR}")
target_link_libraries(mozc-client-cli PRIVATE ${Protobuf_LIBRARIES})

include(GNUInstallDirs)
install(TARGETS mozc-client-cli DESTINATION ${CMAKE_INSTALL_BINDIR})
